## 1ï¸âƒ£ map()

### Purpose

- Transform **each element** of an array
- Always returns a **new array**
- Length of new array = length of original array

### Code Examples

```js
let number = [1, 2, 3, 4, 5];
let newNumber = number.map((ele) => ele + 5);
console.log(newNumber);
```

**Output**

```
[6, 7, 8, 9, 10]
```

---

```js
let number2 = [1, 2, 3, 4, 5];
let newNumber2 = number2.map((ele) => ele * ele);
console.log(newNumber2);
```

**Output**

```
[1, 4, 9, 16, 25]
```

---

```js
let words = ["playwright", "selenium", "cypress"];
let upper = words.map((ele) => ele.toUpperCase());
console.log(upper);
```

**Output**

```
["PLAYWRIGHT", "SELENIUM", "CYPRESS"]
```

---

```js
let words1 = ["playwright", "selenium", "cypress"];
let firstChar = words1.map((ele) => ele[0]);
console.log(firstChar);
```

**Output**

```
["p", "s", "c"]
```

---

```js
let words2 = ["playwright", "selenium", "cypress"];
let conct = words2.map((ele) => "web " + ele);
console.log(conct);
```

**Output**

```

```

### Use Cases

- Convert API response data
- Transform test data
- Create derived values

## 2ï¸âƒ£ filter()

### Purpose

- Select elements based on a **condition**
- Returns a **new array** with fewer or equal elements

### Code Examples

```js
let number3 = [1, 2, 3, 4, 5];
let filterNumber = number3.filter((ele) => ele > 3);
console.log(filterNumber);
```

**Output**

```
[4, 5]
```

---

```js
let number32 = [1, 2, 3, 4, 5];
let evennumbers = number32.filter((ele) => ele % 2 === 0);
console.log(evennumbers);
```

**Output**

```
[2, 4]
```

---

```js
let footer = [
  "amazon help",
  "contact",
  "amazon music",
  "amazon prime",
  "price",
];

let amazonlist = footer
  .filter((ele) => ele.startsWith("amazon"))
  .filter((ele) => ele.includes("music"));

console.log(amazonlist);
```

**Output**

```
["amazon music"]
```

### Use Cases

- Search filters
- Valid test data extraction
- Conditional selection

---

## 3ï¸âƒ£ reduce()

### Purpose

- Reduce array to a **single value**
- Can return number, string, array, object

### Code Examples

```js
let number5 = [1, 2, 3, 4, 5];
let reduceNumber = number5.reduce((result, num) => result + num, 0);
console.log(reduceNumber);
```

**Output**

```
15
```

---

```js
let dict = ["playwright", "selenium", "cypress"];
let newdic = dict.reduce((result, ele) => result + " " + ele);
console.log(newdic);
```

**Output**

```
playwright selenium cypress
```

---

### Combo: map + filter + reduce

```js
let n1 = [85, 92, 67, 78, 95, 43, 88, 76, 91, 82];

let avg =
  n1
    .filter((ele) => ele >= 70)
    .map((ele) => ele + 5)
    .reduce((result, ele) => result + ele, 0) /
  n1.filter((ele) => ele >= 70).length;

console.log(avg);
```

**Output**

```
87.5
```

---

### Use Cases

- Total, average, score calculation
- Report generation
- Data aggregation

## PART 2: Destructuring

Destructuring allows **clean extraction** of values from arrays and objects.

---

## Object Destructuring

```js
let user = {
  name: "hitesh",
  age: 33,
  city: "dehgam",
  ispermenent: true,
};

function getInfo({ name, age }) {
  console.log(name);
  console.log(age);
}

getInfo(user);
```

**Output**

```
hitesh
33
```

---

## Array Destructuring

```js
let products = ["macbook", "keyboard", "imac"];
let [a, b, c] = products;
console.log(a, b, c);
```

**Output**

```
macbook keyboard imac
```

---

### Skip Values

```js
let prod = ["macbook", "keyboard", "imac"];
let [a1, , c1] = prod;
console.log(a1);
console.log(c1);
```

**Output**

```
macbook
imac
```

---

### Default Values

```js
let products33 = ["macbook", "keyboard", "imac"];
let [first = "a", second, third, fourth = "abc"] = products33;
console.log(first, second, third, fourth);
```

**Output**

```
macbook keyboard imac abc
```

---

### Rest Operator

```js
let number = [1, 2, 3, 4, 5];
let [head, ...tail] = number;
console.log(head);
console.log(tail);
```

**Output**

```
1
[2, 3, 4, 5]
```

---

### alias in Object destructuring

```js
// Original object

  name: "Hitesh",

  city: "Dehgam",
  role: "QA Lead",

  console.log(fullName);
```

**Output**

```

QA Lead
```

## Best Practices

- Prefer `map/filter/reduce` over loops for clarity
- Always provide initial value in `reduce`
- Do not mutate original arrays
- Use destructuring in function parameters
- Keep chains readable (avoid very long chains)
